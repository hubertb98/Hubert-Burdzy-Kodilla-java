SELECT IL.NAME AS LIST_NAME, I.SUMMARY AS TASK_SUMMARY
FROM ISSUESLIST IL
JOIN ISSUESS I ON IL.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY AS TASK_SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUESS I
JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS POSTS_NUMBER
FROM USERS U JOIN ISSUESS I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID;
